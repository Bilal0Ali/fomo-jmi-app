rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    match /resources/{userId}/{fileName} {
      // Allow authenticated users to read any file in the 'resources' path
      allow read: if request.auth != null;
      // Allow authenticated users to write files only to their own folder
      allow write: if request.auth != null && request.auth.uid == userId;
    }
    match /resources/{fileName} {
       // Allow authenticated users to read any file in the 'resources' root (if you store files directly there)
       // If all files are in user subfolders, this might not be needed or you might restrict it further.
       allow read: if request.auth != null;
       // Disallow writing to the root 'resources' path to enforce user subfolders for writes
       allow write: if false;
    }
     // Deny read/write to any other paths by default
    match /{allPaths=**} {
      allow read, write: if false;
    }
  }
}
